%%%
%%% Scenario:
%%% One conveyor (C1) connected to a railway (R1).
%%% The conveyor has an RFID Actor (RF1) connected to it.
%%%

C1 = spawn(container, idling, [actor_conveyor:create()]).
R1 = spawn(container, idling, [actor_railway:create()]).
RF1 = spawn(container, idling, [actor_rfid:create()]).
C1 ! R1 ! RF1 ! {start}.

C1 ! {self(), {add, out, R1}}.
C1 ! {self(), {add, option, {rfid, RF1}}}.

C1 ! {self(), {actor_product, actor_product:create()}}.
